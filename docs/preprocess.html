happy_sorted = happy.sort_values(by='country')

merged_df = pd.DataFrame()  


threshold = 30

for i, row1 in happy_sorted.iterrows():
    max_similarity = 0
    matching_row = None
    for j, row2 in agg_health.iterrows():
        country2 = re.sub(r'\(.*\)', '', row2.name).strip() 
        similarity = fuzz.token_sort_ratio(row1.iloc[0], country2)
        if similarity > max_similarity:
            max_similarity = similarity
            matching_row = row2
    if max_similarity > threshold: 
        merged_row = pd.concat([row1, matching_row])
        merged_df = merged_df.append(merged_row, ignore_index=True)
